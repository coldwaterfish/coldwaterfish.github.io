##### 基本配置

```js
var mCharts=echarts.init(dom)
$.get('url',function(res){
    echarts.registerMap('chinaMap',res)
    var option={
        geo:{
            type:'map',
            // 需要和registerMap保持一致
            map:'chinaMap'
        }
    }
    mcharts.setOption(option)
})
```

#### 显示某个区域

```js
加载该区域的矢量地图数据
registerMap注册
指明geo配置下的type和map属性
通过zoom放大该区域
通过center定位中心点
```

```js
echarts.registerMap('anhui',res)
var option={
    geo:{
        type:'map',
        // 需要和registerMap保持一致
        map:'anhui',
        zoom:1.2,//缩放
        label:{
            show:true
        },
        center:[经度,纬度],//通过矢量数据获取到
    }
}
mcharts.setOption(option)
```

#### 不同颜色

```js
geo配合series
关联geo和series数据
结合visualMap配合
```

```js
var airData=[
    {name:'北京',value:},
     ...
]
var option={
     geo:{...}
     series:[
        {
            type:'map',
            data:airData,
            geoIndex:0  // 关联到第一个地图
        }
     ],
     visualMap:{
         min:0,
         max:300,
         inRange:{
             // 颜色渐变
             color:['white','red']
         },
         // 筛选
         calculable:true
     }
}
```

#### 结合散点图

```js
在series下增加新的对象
散点数据设置给新对象的data
新对象的type:effectScatter
散点图使用地图坐标系统：coordinateSystem:'geo'
让涟漪效果更加明显
rippleEffect:{scale:10}
```

```js
var scatterData=[
    {
        value:[117.283042,31.86119]
    }
]
series:[
    {
        type:'map',
        data:airData,
        geoIndex:0  // 关联到第一个地图
    },
    {
        type:effectScatter,
        data:scatterData,
        coordinateSystem:'geo',
        rippleEffect:{scale:10}
    }
],
```
